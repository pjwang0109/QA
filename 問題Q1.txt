function validateFile(input) {
    const file = input.files[0];
    const errorSpan = $(input).next('.error-message');

    if (file && !file.name.toLowerCase().endsWith('.xlsx')) {
        errorSpan.text('請上傳 .xlsx 檔案');
        input.value = ''; // 清空不合格的檔案
    } else {
        errorSpan.text('');
    }
}
