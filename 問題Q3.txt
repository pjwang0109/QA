  <table class="table table-bordered" id="product-table">
    <thead>
      <tr id="table-header">
        <th>產品名稱</th>
        <th>佔營收比</th>
        <!-- 年度欄位插入位置 -->
      </tr>

      <!-- 產品資料列插入位置 -->

<script src="https://code.jquery.com/jquery-3.6.0.min.js"></script>
<script>
  // 狀態變數
  let yearList = [];
  const maxYears = 3;
  let nextYear = new Date().getFullYear();

  // 初始化
  $(document).ready(function () {
    $('#add-year-btn').click(addYearColumn);
    $('#add-product-btn').click(addProductRow);
    $(document).on('click', '.remove-year-btn', removeYearColumn);
  });

  // ✅ 新增一個年度欄位
  function addYearColumn() {
    if (yearList.length >= maxYears) {
      alert('最多只能新增 3 個年份');
      return;
    }

    const year = nextYear++;
    yearList.push(year);

    // 新增表頭欄位
    $('#table-header').append(`
      <th class="year-col-${year}">
        ${year}
        <button class="btn btn-xs btn-danger remove-year-btn" data-year="${year}">
          <i class="glyphicon glyphicon-remove"></i>
        </button>
      </th>
    `);

    // 為現有的每一列產品加上金額輸入欄位
    $('#product-body .product-row').each(function () {
      $(this).append(`
        <td class="year-col-${year}">
          <input type="text" class="form-control amount" placeholder="金額">
        </td>
      `);
    });
  }

  // ✅ 移除指定年份欄位與資料欄
  function removeYearColumn() {
    const year = $(this).data('year');
    $(`.year-col-${year}`).remove();
    yearList = yearList.filter(y => y !== year);
  }

  // ✅ 新增一筆產品資料列
  function addProductRow() {
    const $row = $('<tr class="product-row"></tr>');

    // 基本欄位
    $row.append(`
      <td><textarea class="form-control pname" rows="1" placeholder="產品名稱"></textarea></td>
      <td>
        <div class="input-group">
          <input type="text" class="form-control rate" maxlength="3" placeholder="%" />
          <span class="input-group-addon">%</span>
        </div>
      </td>
    `);

    // 根據目前的年份欄位數新增金額欄
    yearList.forEach(year => {
      $row.append(`
        <td class="year-col-${year}">
          <input type="text" class="form-control amount" placeholder="金額">
        </td>
      `);
    });

    $('#product-body').append($row);
  }
</script>
