function setPageDataByJsonClassNameMappingCR(jsondata, $div) {
    $.each(jsondata, (k, v) => {
if ($input.is("select")) {
            $div.find(`select.${k}`).val(v);
            $div.find(`select.${k}`).data("selected", v);
        }           

else if ($input.is("input:checkbox")) {
            console.log($input);
            let vArr = [];
            if (v.indexOf(",") != -1) {
                vArr = v.split(",");
            } else {
                vArr.push(v);
            }
            if (v && vArr.length > 0) {
                $.each(vArr, (i, val) => {
                    $div.find(`.${k}[value=${val}]`).prop("checked", true).change();
                });
            }

})
}
